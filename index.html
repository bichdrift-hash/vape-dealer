function copyOrderText() { const orderText = document.getElementById('order-text-area').value; // Проверяем, открыт ли сайт внутри Telegram WebApp if (window.Telegram && window.Telegram.WebApp) { const tg = window.Telegram.WebApp; // Отправляем данные заказа обратно боту tg.sendData(orderText); tg.close(); // Закрываем мини-приложение после отправки } else { // Если открыли просто в обычном браузере, сработает обычное копирование const textArea = document.getElementById('order-text-area'); textArea.select(); try { document.execCommand('copy'); alert('Текст заказа скопирован!'); } catch (err) { alert('Не удалось скопировать.'); } } }
